        //由于String类型是 final char[] 的形式存在的，函数内部的String改变是直接开辟新的字符串。
        // 然后将形参的引用指向新开辟的字符串String,将不会影响外部的值的。
        //此处的赋值 是新开辟字符串"" 然后将形参引用s指向它
        s = "";
    }
